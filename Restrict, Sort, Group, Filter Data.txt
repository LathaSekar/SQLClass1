3️⃣ Operators
Comparison Operators
sqlSELECT * FROM employees WHERE salary = 75000;   -- Equal
SELECT * FROM employees WHERE salary != 75000;  -- Not equal
SELECT * FROM employees WHERE salary > 60000;   -- Greater than
SELECT * FROM employees WHERE salary < 60000;   -- Less than
SELECT * FROM employees WHERE salary >= 60000;  -- Greater or equal
SELECT * FROM employees WHERE salary <= 60000;  -- Less or equal
Logical Operators
sql-- AND — Both conditions must be true
SELECT emp_name, department, salary
FROM employees
WHERE department = 'IT' AND salary > 80000;
-- Result: Charlie (90000), Eve (82000)

-- OR — At least one condition must be true
SELECT emp_name, department
FROM employees
WHERE department = 'HR' OR department = 'Marketing';
-- Result: Bob, Frank, Henry, Isla

-- NOT — Reverses the condition
SELECT emp_name, is_active
FROM employees
WHERE NOT is_active = true;
-- Result: Diana, Henry (inactive employees)
Special Operators
sql-- BETWEEN (inclusive range)
SELECT emp_name, salary FROM employees
WHERE salary BETWEEN 50000 AND 75000;

-- IN (match any value in list)
SELECT emp_name, city FROM employees
WHERE city IN ('New York', 'Chicago');

-- LIKE (pattern matching)
-- %  → any number of characters
-- _  → exactly one character
SELECT emp_name FROM employees WHERE emp_name LIKE 'A%';     -- Starts with A
SELECT emp_name FROM employees WHERE emp_name LIKE '__a%';   -- 3rd letter is 'a'

-- IS NULL / IS NOT NULL
SELECT emp_name FROM employees WHERE city IS NOT NULL;
```
4️⃣ Precedence of Operators

> SQL evaluates operators in this order (like BODMAS in math):
```
1. ()  → Parentheses (highest priority)
2. *,/,% → Arithmetic
3. +,-   → Arithmetic
4. =, !=, >, <, >=, <=, LIKE, IN, BETWEEN, IS NULL
5. NOT
6. AND
7. OR   (lowest priority)
⚠️ Precedence Trap — Wrong vs Right
sql-- ❌ WRONG — AND runs before OR (unintended grouping)
SELECT * FROM employees
WHERE department = 'IT' OR department = 'HR' AND salary > 60000;
-- Interpreted as: IT  OR  (HR AND salary > 60000)
-- Gets ALL IT employees + only high-earning HR employees

-- ✅ CORRECT — Use parentheses to control order
SELECT * FROM employees
WHERE (department = 'IT' OR department = 'HR') AND salary > 60000;
-- Gets IT and HR employees who earn more than 60000
Real-Time Precedence Example
sql-- Find active employees from IT or Finance earning over 70k
SELECT emp_name, department, salary, is_active
FROM employees
WHERE is_active = true
  AND (department = 'IT' OR department = 'Finance')
  AND salary > 70000;

-- Result:
-- Alice   | IT      | 75000
-- Charlie | IT      | 90000
-- Eve     | IT      | 82000
-- Grace   | Finance | 71000
-- Jack    | Finance | 85000

5️⃣ ORDER BY — Sorting Data

Sorts the final result. Applied after WHERE, GROUP BY, HAVING.

sql-- ASC (default) — Lowest to Highest
SELECT emp_name, salary FROM employees
ORDER BY salary ASC;

-- DESC — Highest to Lowest
SELECT emp_name, salary FROM employees
ORDER BY salary DESC;

-- Sort by multiple columns
SELECT emp_name, department, salary
FROM employees
ORDER BY department ASC, salary DESC;
-- Groups by dept alphabetically, then sorts salary high→low within each dept

-- Sort by column position (2 = salary column)
SELECT emp_name, salary FROM employees
ORDER BY 2 DESC;

-- Real-world: Get Top 3 highest paid employees
SELECT emp_name, salary FROM employees
ORDER BY salary DESC
LIMIT 3;
-- Result: Charlie (90000), Jack (85000), Eve (82000)

6️⃣ GROUP BY — Grouping Data

Collapses rows into groups. Used with aggregate functions.

Aggregate Functions
sqlCOUNT()  -- Count rows
SUM()    -- Total of values
AVG()    -- Average value
MAX()    -- Highest value
MIN()    -- Lowest value
GROUP BY Examples
sql-- Count employees per department
SELECT department, COUNT(*) AS total_employees
FROM employees
GROUP BY department;

-- Result:
-- IT        | 3
-- HR        | 2
-- Finance   | 3
-- Marketing | 2

-- Total salary spent per department
SELECT department, SUM(salary) AS total_salary
FROM employees
GROUP BY department;

-- Average salary per department
SELECT department,
       ROUND(AVG(salary), 2) AS avg_salary,
       MAX(salary)           AS highest_salary,
       MIN(salary)           AS lowest_salary
FROM employees
GROUP BY department;

-- Count active vs inactive employees per city
SELECT city, is_active, COUNT(*) AS count
FROM employees
GROUP BY city, is_active
ORDER BY city;

7️⃣ HAVING — Filter Groups

Like WHERE, but filters after grouping. Works on aggregated data.

sql-- WHERE  → filters rows   (before grouping)
-- HAVING → filters groups (after grouping)
Basic HAVING
sql-- Departments with more than 2 employees
SELECT department, COUNT(*) AS total
FROM employees
GROUP BY department
HAVING COUNT(*) > 2;

-- Result:
-- IT      | 3
-- Finance | 3

-- Departments with average salary above 65000
SELECT department, ROUND(AVG(salary), 2) AS avg_salary
FROM employees
GROUP BY department
HAVING AVG(salary) > 65000;

-- Result:
-- IT      | 82333.33
-- Finance | 72000.00
HAVING vs WHERE Together
sql-- Step 1: WHERE filters active employees only
-- Step 2: GROUP BY groups them by department
-- Step 3: HAVING filters groups with total salary > 150000

SELECT department,
       COUNT(*)        AS active_count,
       SUM(salary)     AS total_salary,
       AVG(salary)     AS avg_salary
FROM employees
WHERE is_active = true          -- ✅ Filters rows FIRST
GROUP BY department
HAVING SUM(salary) > 150000    -- ✅ Filters groups AFTER
ORDER BY total_salary DESC;

-- Result:
-- IT      | 3 | 247000 | 82333
-- Finance | 2 | 156000 | 78000
8️⃣ Full Query — All Clauses Together
sql-- Business Question:
-- "Which active departments (excluding Marketing) have more than 1 employee,
--  average salary over 60000? Sort by avg salary highest first."

SELECT
    department,
    COUNT(*)               AS total_employees,
    ROUND(AVG(salary), 2)  AS avg_salary,
    SUM(salary)            AS total_cost,
    MAX(salary)            AS top_salary
FROM employees
WHERE is_active = true                -- 1️⃣ Filter rows
  AND department != 'Marketing'       -- 1️⃣ Exclude Marketing
GROUP BY department                   -- 2️⃣ Group
HAVING COUNT(*) > 1                   -- 3️⃣ Filter groups
   AND AVG(salary) > 60000            -- 3️⃣ Filter groups
ORDER BY avg_salary DESC;             -- 4️⃣ Sort

-- Result:
-- IT      | 3 | 82333.33 | 247000 | 90000
-- Finance | 2 | 78000.00 | 156000 | 85000
-- HR      | 2 | 46500.00 filtered out (avg < 60000)

SQL EXISTS —
📌 What is EXISTS?
sql-- EXISTS checks whether a subquery returns ANY rows
-- Returns TRUE  → if subquery has at least 1 row
-- Returns FALSE → if subquery returns no rows

SELECT column
FROM table
WHERE EXISTS (SELECT 1 FROM another_table WHERE condition);

EXISTS stops scanning as soon as it finds the first matching row — making it very fast!

1️⃣ Basic EXISTS — Customers WHO placed orders
sqlSELECT customer_name, city
FROM customers c
WHERE EXISTS (
    SELECT 1
    FROM orders o
    WHERE o.customer_id = c.customer_id  -- correlated condition
);

-- Result:
-- Alice   | New York
-- Bob     | Chicago
-- Charlie | Houston
-- ✅ Diana and Eve are excluded (no orders)

💡 SELECT 1 is used inside EXISTS because we only care if rows exist, not what they contain.


2️⃣ NOT EXISTS — Customers WHO never placed orders
sqlSELECT customer_name, city, email
FROM customers c
WHERE NOT EXISTS (
    SELECT 1
    FROM orders o
    WHERE o.customer_id = c.customer_id
);

-- Result:
-- Diana | New York    | diana@gmail.com
-- Eve   | Los Angeles | eve@gmail.com
Real-world use: Send a promo email to customers who have never ordered.

6️⃣ EXISTS vs IN — Key Difference
sql-- Using IN
SELECT customer_name
FROM customers
WHERE customer_id IN (
    SELECT customer_id FROM orders
);

-- Using EXISTS (equivalent but faster on large data)
SELECT customer_name
FROM customers c
WHERE EXISTS (
    SELECT 1 FROM orders o
    WHERE o.customer_id = c.customer_id
);
7️⃣ EXISTS with Multiple Conditions

9️⃣ EXISTS in UPDATE — Real-world use case
sql-- Add a 'has_orders' flag column
ALTER TABLE customers ADD COLUMN has_orders BOOLEAN DEFAULT false;

-- Update flag to TRUE for customers who have orders
UPDATE customers c
SET has_orders = true
WHERE EXISTS (
    SELECT 1
    FROM orders o
    WHERE o.customer_id = c.customer_id
);

-- customers table now:
-- Alice   | has_orders = true
-- Bob     | has_orders = true
-- Charlie | has_orders = true
-- Diana   | has_orders = false ← unchanged
-- Eve     | has_orders = false ← unchanged

🔟 EXISTS in DELETE — Clean up orphan records
sql-- Delete customers who have NO orders (cleanup inactive accounts)
DELETE FROM customers
WHERE NOT EXISTS (
    SELECT 1
    FROM orders o
    WHERE o.customer_id = customers.customer_id
);

-- Deletes: Diana and Eve
